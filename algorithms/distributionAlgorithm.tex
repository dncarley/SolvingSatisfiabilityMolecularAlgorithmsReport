\begin{pseudocode}{Insert Variable}{T, v}
	T_R \GETS \emptyset \\
	\FOREACH \text{ string } s \text{ in } T \DO
		\BEGIN
			\CASE(1): v < s \THEN
				s' \GETS \text{append}(v, s)\\
			\CASE(2): v > s \THEN
				s' \GETS \text{append}(s, v)\\
			\CASE(3): s_1 < v < s_2 \THEN
				\BEGIN
				[s_1, s_2] \GETS \text{splice}(s, v)\\
				s_1 \GETS \text{append}(s_1, v)\\
				s' \GETS \text{append}(s_1, s_2)\\
				\END \\
			\CASE(4): \neg v \in s \THEN
				s' \GETS \emptyset \\
			\CASE(5): v \in s \THEN
				s' \GETS s\\
			T_R \GETS \text{mix}(T_R, s') \\
		\END \\
	\RETURN{T_R}
\end{pseudocode}

		\END\\

	\END
