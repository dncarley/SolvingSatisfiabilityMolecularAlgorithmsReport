


\begin{pseudocode}{Combinatorial Generate}{n}
	T_{comb} \GETS \emptyset \\
	T_{comb} \GETS \text{mix(}T_{comb}, start\text{)} \\ 

	\FOR i \GETS 1 \text{ to } n \DO
		\BEGIN
		
		[T_1,T_2] \GETS \text{split(} T_{comb}\text{)}\\
		T_1 \GETS \text{append(}T_1, POS \text{)}\\
		T_2 \GETS \text{append(}T_2, NEG \text{)}\\
		T_{comb} \GETS \text{mix(}T_1,T_2\text{)}\\
	\END
	\\
	\RETURN {T_{comb}}
\end{pseudocode}

\begin{pseudocode}{Lipton's Algorithm}{\phi}
T \GETS \text{{\sc Combinatorial Generate}}( \mid \phi \mid) \\
\FOREACH \text{clause } C \text{ in } \phi \DO
	\BEGIN
	T_c \GETS \emptyset \\
	\FOREACH \text{variable } v \text{ in } C \DO
		\BEGIN
			\IF v \text{ is a positive literal} \THEN
				\BEGIN
					T_P \GETS \text{extract}(T, POS, v)\\
					T_c \GETS \text{mix}(T_P, T_c)						
				\END
			\ELSE
				\BEGIN
					T_N \GETS \text{extract}(T, NEG, v)\\
					T_c \GETS \text{mix}(T_N, T_c)						
				\END
		\END
	\\
	T \GETS T_c\\
	\END
\\
\RETURN{\text{detect}(T)}
\end{pseudocode}
